1. Дан код:

var a = 1, b = 1, c, d;

c = ++a; alert(c);           // 2
Первая операция - инкремент переменной "а": было 1, стало 2
Вторая операция - присвоение переменной "с" инкрементированного значения "а": = 2
Печать значения переменной "с" = 2

d = b++; alert(d);           // 1
Первая операция - присвоение переменной "d" значения переменной "b": = 1
Вторая операция - инкремент переменной "b": было 1, стало 2
Печать значения переменной "d" = 1


c = (2+ ++a); alert(c);      // 5
Первая операция - инкремент переменной "а": было 2, стало 3
Вторая операция - сложение числа 2 с инкрементированным значением "а": 2 + 3 = 5
Третяя операция - присвоение переменной "с" результата суммирования: = 5
Печать значения переменной "с" = 5

d = (2+ b++); alert(d);      // 4
Первая операция - сложение числа 2 с переменной "b": 2 + 2 = 4
Вторая операция - инкремент переменной "b": было 2, стало 3
Третяя операция - присвоение переменной "d" результата суммирования: = 5
Печать значения переменной "с" = 5

alert(a);                    // 3
